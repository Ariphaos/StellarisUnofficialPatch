			trigger = {
				has_food_upkeep = yes
			}

		}

	weight = {
		weight = @colonist_job_weight # Higher prio

			factor = 1.5

		}
		modifier = {
			factor = 0
			NOT = { has_job = colonist }
			exists = planet

				owner = { is_ai = yes }
				count_owned_pop = { limit = { is_sapient = yes }count < 4 }

					has_job = colonist
				}

		}
		modifier = {
