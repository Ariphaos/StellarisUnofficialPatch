					}
				}
			}
			planet_event = { id = astral_planes.2120 }
		}

		if = {
			limit = {
				OR = {
					has_valid_civic = civic_dimensional_worship
					has_valid_civic = civic_corporate_dimensional_worship

			}
			random_owned_starbase = {
				set_starbase_building = {
					slot = 1
					building = dimensional_shrine
				}
			}
		}
	}
}
