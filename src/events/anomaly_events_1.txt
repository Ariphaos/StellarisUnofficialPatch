	}
}

	option = {

		trigger = {

					# Commonwealth of Man NPC system

			}
		}

		}

				value = @tier3researchreward

				max = @tier3researchmax
			}
			begin_event_chain = {
				event_chain = "find_sol_chain"
				target = this
			}
		}
		hidden_effect = {
			random_system = {
				limit = {
					OR = {
						# Commonwealth of Man NPC system
						has_star_flag = com_sol
