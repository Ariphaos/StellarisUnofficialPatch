			has_job_category = worker

	triggered_upkeep_modifier = {
		key = planet_jobs_worker
		modifier_types = {
			mult

		key = planet_jobs_worker
		modifier_types = {
			mult
			add
		}
		trigger = {
			OR = {
				is_pop_category = worker
				AND = {
					is_pop_category = slave # Yes, slave should count towards both worker and slave
					has_job_category = worker
				}
				is_pop_category = simple_drone
			}
		}
	}

		}

					is_pop_category = slave # Yes, slave should count towards both worker and slave
					has_job_category = worker
