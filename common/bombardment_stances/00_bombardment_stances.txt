selective = {
	trigger = {
		OR = {
			owner = {
				is_ai = no
				OR = {
					has_policy_flag = orbital_bombardment_selective
					has_policy_flag = orbital_bombardment_indiscriminate
					has_policy_flag = orbital_bombardment_armageddon
				}
			}
			owner = {
				is_ai = yes
				has_policy_flag = orbital_bombardment_selective
				NOR = {
					has_policy_flag = orbital_bombardment_indiscriminate
					has_policy_flag = orbital_bombardment_armageddon
					has_valid_civic = civic_barbaric_despoilers
					has_ascension_perk = ap_nihilistic_acquisition
				}
			}
		}
	}

	icon_frame = 1
	default = yes

	stop_when_armies_dead = yes
	abduct_pops = no

	planet_damage = 0.5
	army_damage = 1.0

	kill_pop_chance = {
		base = 0.25
	}
	min_pops_to_kill_pop = 21

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 1
	}
}

indiscriminate = {
	trigger = {
		owner = {
			OR = {
				NOT = { is_country_type = default }
				has_policy_flag = orbital_bombardment_indiscriminate
				has_policy_flag = orbital_bombardment_armageddon
			}
			OR = {
				is_ai = no
				AND = {
					NOT = { has_policy_flag = orbital_bombardment_armageddon }
					NOT = { owner_species = { has_trait = trait_pc_nuked_preference } }
					NOT = { owner_species = { has_trait = trait_survivor } }
				}
			}
		}
	}

	icon_frame = 2
	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 1.0
	army_damage = 1.5

	kill_pop_chance = {
		base = 0.5
	}
	min_pops_to_kill_pop = 11

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 5
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
		modifier = {
			factor = 0
			exists = from
			from = { owner = { NOT = { is_hostile = root.owner } } }
			owner = { 
				has_policy_flag = orbital_bombardment_armageddon
				has_valid_civic = civic_machine_terminator
			}
		}
	}
}

armageddon = {
	trigger = {
		owner = {
			OR = {
				has_policy_flag = orbital_bombardment_armageddon
				is_crisis_faction = yes
			}
		}
	}

	icon_frame = 3
	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 1.5
	army_damage = 2.0

	kill_pop_chance = {
		base = 1.0
	}
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
			}
			exists = owner
			owner = { is_crisis_faction = no }
		}
	}
}

pox = {
	trigger = {
		owner = {
			has_modifier = javorian_pox
			OR = {
				NOT = { is_country_type = default }
				has_policy_flag = orbital_bombardment_indiscriminate
				has_policy_flag = orbital_bombardment_armageddon
			}
		}
	}

	icon_frame = 5
	default = no

	stop_when_armies_dead = no
	abduct_pops = no

	planet_damage = 0.2
	army_damage = 1.5

	kill_pop_chance = { # Should only target organics
		base = 1.5
		modifier = {
			factor = 0
			OR = {
				pop_has_trait = trait_machine_unit
				pop_has_trait = trait_mechanical
			}
		}
	}
	min_pops_to_kill_pop = 0

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
	}
}

raiding = {
	trigger = {
		OR = {
			owner = { has_ascension_perk = ap_nihilistic_acquisition }
			owner = { has_valid_civic = civic_barbaric_despoilers }
		}
	}

	icon_frame = 4
	default = no

	stop_when_armies_dead = no
	abduct_pops = yes

	planet_damage = 1.0
	army_damage = 0.5

	kill_pop_chance = {
		base = 1.0
	}
	min_pops_to_kill_pop = 2

	# root = fleet
	# from = planet
	ai_weight = {
		weight = 8
		modifier = {
			factor = 0
			exists = from
			from = {
				owner = { NOT = { is_hostile = root.owner } }
			}
		}
		modifier = {
			factor = 0.01
			#exists = from
			root = {
				owner = { has_claim = root.solar_system }
			}
		}
	}
}
