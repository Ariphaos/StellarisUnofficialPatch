building_military_academy = {
	base_buildtime = @b2_time
	base_cap_amount = 1

	category = army

	potential = {
		buildings_simple_allow = yes
		exists = owner
		owner = { NOT = { has_valid_civic = civic_reanimated_armies } }
	}

	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				is_ai = yes
				has_valid_civic = civic_reanimated_armies
			}
		}
	}

	convert_to = {
		building_dread_encampment
	}

	allow = {
		has_upgraded_capital = yes
	}

	army_modifier = {
		army_starting_experience_add = 100
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = @b1_jobs
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = @b1_jobs
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
		}
	}

	prerequisites = {
		"tech_centralized_command"
	}

	show_tech_unlock_if = {
		NOT = { has_valid_civic = civic_reanimated_armies }
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}
}

building_stronghold = {
	base_buildtime = 240
	position_priority = 100

	category = army

	potential = {
		buildings_simple_allow = yes
	}

	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		AND = {
			OR = {
				num_buildings = { type = building_stronghold value > 1 }
				has_building = building_fortress
			}
			free_building_slots = 0
			owner = {
				used_naval_capacity_percent < 0.6
				is_at_war = no # don't destroy while at war
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = @b1_jobs
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = @b1_jobs
		}
	}

	resources = {
		category = planet_buildings_strongholds
		produces = {
			trigger = {
				exists = owner
				owner = {
					has_active_tradition = tr_unyielding_resistance_is_frugal
				}
			}
			unity = 3
		}
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 1
		}
	}

	prerequisites = {
		"tech_planetary_defenses"
	}

	upgrades = {
		"building_fortress"
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {

			exists = owner

		}
		text = job_soldier_effect_desc
	}
}

building_fortress = {
	base_buildtime = @b1_time
	can_build = no
	planetary_ftl_inhibitor = yes
	position_priority = 100

	category = army

	allow = {
		has_upgraded_capital = yes
		exists = owner
		OR = {
			owner = { is_ai = no }
			AND = {
				buildings_upgrade_allow = yes
				buildings_has_motes = yes
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
		OR = {
			AND = {
				num_buildings = { type = building_fortress value > 1 }
				free_building_slots = 0

			}

		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}

			job_warrior_drone_add = @b2_jobs
		}
	}

	resources = {

	prerequisites = {

	triggered_desc = {

		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}
}
