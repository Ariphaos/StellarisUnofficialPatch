agenda_slave_optimization = {
	# Slave output bonus
	weight_modifier = {
		weight = 100
		modifier = {
			factor = 1.5
			from = {
				has_ethic = ethic_fanatic_authoritarian
			}
		}
		modifier = {
			factor = 2
			exists = pop_faction
			pop_faction = { is_pop_faction_type = totalitarian }
		}
		modifier = {
			factor = 0
			from = {
				OR = {
					has_policy_flag = slavery_not_allowed
					AND = {
						check_modifier_value = {
							modifier = country_pop_enslaved_mult
							value = 0
						}
						count_enslaved_species = { count = 0 }
					}
				}
			}
		}
		modifier = {
			factor = 0
			years_passed < 1 # do non-ais have a valid ai personality?
			from = {
				NOT = { has_policy_flag = slavery_allowed }
				is_authoritarian = no # Required for slaver guilds / indentured assets
				is_xenophobe = no
			}
		}
	}

	modifier = {
		planet_jobs_slave_produces_mult = 0.10
	}
}
